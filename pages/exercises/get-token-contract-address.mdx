# Get token contract address

The token contract address is used to retrieve the token contract.

Open the project in your IDE and edit `src/App.js`:

```js filename="src/App.js" {17-35,41-44} showLineNumbers copy
import AElf from "aelf-sdk";
import { useState } from "react";

// create a new instance of AElf
const aelf = new AElf(
  new AElf.providers.HttpProvider("https://aelf-test-node.aelf.io")
);

function App() {
  const [wallet, setWallet] = useState();
  const [tokenContractAddress, setTokenContractAddress] = useState();

  const generateWallet = () => {
    setWallet(AElf.wallet.createNewWallet());
  };

  const getTokenContractAddress = async () => {
    const tokenContractName = "AElf.ContractNames.Token";
    let tokenContractAddress;
    // get chain status
    const chainStatus = await aelf.chain.getChainStatus();
    // get genesis contract address
    const GenesisContractAddress = chainStatus.GenesisContractAddress;
    // get genesis contract instance
    const zeroContract = await aelf.chain.contractAt(
      GenesisContractAddress,
      wallet
    );
    // Get contract address by the read only method `GetContractAddressByName` of genesis contract
    tokenContractAddress = await zeroContract.GetContractAddressByName.call(
      AElf.utils.sha256(tokenContractName)
    );

    setTokenContractAddress(tokenContractAddress);
  };

  return (
    <div>
      <div>Wallet address: {wallet ? wallet.address : ""}</div>
      <button onClick={generateWallet}>Generate new wallet</button>
      <div>Token contract address: {tokenContractAddress}</div>
      <button onClick={getTokenContractAddress}>
        Get token contract address
      </button>
    </div>
  );
}

export default App;
```

- Visit http://localhost:3000 in your browser.
- Click on the "Generate new wallet" button.
- Observe that the wallet address changes.
- Click on the "Get token contract address" button.
- Observe that the token contract address changes.

Congratulations! You have retrieved the token contract address.

# Call GetBalance method in token contract

Now that you have the contract address, you may use it to call the GetBalance viewMethod to retrieve the balance from the chain.

Open the project in your IDE and edit `src/App.js`:

```js filename="src/App.js" {12,38-51,61-62} showLineNumbers copy
import AElf from "aelf-sdk";
import { useState } from "react";

// create a new instance of AElf
const aelf = new AElf(
  new AElf.providers.HttpProvider("https://aelf-test-node.aelf.io")
);

function App() {
  const [wallet, setWallet] = useState();
  const [tokenContractAddress, setTokenContractAddress] = useState();
  const [balance, setBalance] = useState("");

  const generateWallet = () => {
    setWallet(AElf.wallet.createNewWallet());
  };

  const getTokenContractAddress = async () => {
    const tokenContractName = "AElf.ContractNames.Token";
    let tokenContractAddress;
    // get chain status
    const chainStatus = await aelf.chain.getChainStatus();
    // get genesis contract address
    const GenesisContractAddress = chainStatus.GenesisContractAddress;
    // get genesis contract instance
    const zeroContract = await aelf.chain.contractAt(
      GenesisContractAddress,
      wallet
    );
    // Get contract address by the read only method `GetContractAddressByName` of genesis contract
    tokenContractAddress = await zeroContract.GetContractAddressByName.call(
      AElf.utils.sha256(tokenContractName)
    );

    setTokenContractAddress(tokenContractAddress);
  };

  const getBalance = async () => {
    const tokenContract = await aelf.chain.contractAt(
      tokenContractAddress,
      wallet
    );
    // get the balance of an address, this would not send a transaction,
    // or store any data on the chain, or required any transaction fee, only get the balance
    // with `.call` method, `aelf-sdk` will only call read-only method
    const result = await tokenContract.GetBalance.call({
      symbol: "ELF",
      owner: wallet.address,
    });
    setBalance(JSON.stringify(result));
  };

  return (
    <div>
      <div>Wallet address: {wallet ? wallet.address : ""}</div>
      <button onClick={generateWallet}>Generate new wallet</button>
      <div>Token contract address: {tokenContractAddress}</div>
      <button onClick={getTokenContractAddress}>
        Get token contract address
      </button>
      <div>Balance: {balance}</div>
      <button onClick={getBalance}>Get balance</button>
    </div>
  );
}

export default App;
```
